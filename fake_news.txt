<script>
function checkNews() {

    let text = document.getElementById("newsText").value.toLowerCase();

    // Fake news logic keywords
    const fakeKeywords = [
        "100% cure",
        "miracle",
        "shocking truth",
        "they don't want you to know",
        "guaranteed money",
        "click here fast",
        "secret hack"
    ];

    let score = 0;
    let found = [];

    for (let i = 0; i < fakeKeywords.length; i++) {
        if (text.includes(fakeKeywords[i])) {
            score++;
            found.push(fakeKeywords[i]);
        }
    }

    let result = "";

    if (score >= 2) {
        result = "🚨 Fake News Likely";
    } else if (score == 1) {
        result = "⚠️ Suspicious Content";
    } else {
        result = "✅ Likely Real / Safe";
    }

    document.getElementById("resultBox").innerHTML =
        "<h2>" + result + "</h2>" +
        "<p><b>Score:</b> " + score + "</p>" +
        "<p><b>Keywords Found:</b> " + (found.length ? found.join(", ") : "None") + "</p>";
}
</script>
